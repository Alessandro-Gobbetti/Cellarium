3   /**
4    * DIV divides the top two values from the OperandStack
5    * and pushes the result back to the OperandStack.
6    */
7   public class DIV extends Instruction {
8       
9       @Override
10      public void execute(final Storage storage) {
11          final OperandStack stack = storage.getOperandStack();
12          final double operand1 = stack.pop();
13          final double operand2 = stack.pop();
14          stack.push(operand2 / operand1);
15      }
16      
17      @Override
18      public String toString() {
19          return "DIV";
20      }
21      
22  }
