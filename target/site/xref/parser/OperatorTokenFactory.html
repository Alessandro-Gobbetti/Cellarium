3   /**
4    * A special kind of StringTokenFactory,
5    * which produces tokens that represent operators.
6    */
7   public class OperatorTokenFactory extends StringTokenFactory {
8   
9       private final TokenType tokenType;
10  
11      
12      /**
13       * Create an OperatorTokenFactory for tokens representing the given operator.
14       * @param operator the operator
15       * @param tokenType the TokenType corresponding to this operator
16       */
17      public OperatorTokenFactory(final String operator, final TokenType tokenType) {
18          super(operator);
19          this.tokenType = tokenType;
20      }
21  
22      @Override
23      public Token getToken() {
24          // return a token of this object's TokenType 
25          // with its text and starting position
26          return new Token(tokenType, getTokenText(), getTokenStartPosition());
27      }
28  
29  }
